### Headers

<ResponseField name="Authorization" type="string" required>
  Example:

  `Authorization | Bearer Paste-Your-API-Key-Here`
</ResponseField>

### Body

<ParamField path="licenseCode" type="string" required>
  The license code to be activated.
</ParamField>

<ParamField path="productId" type="string" required>
  The ID of the product associated with the license.
</ParamField>

### Root

<ParamField path="statusCode" type="number">
  HTTP status code (example: `200`).
</ParamField>

<ParamField path="isLicenseActive" type="boolean">
  Indicates whether the license is currently active (`true` or `false`).
</ParamField>

<ParamField path="licenseCode" type="object">
  License code details.
</ParamField>

### licenseCode

<ParamField path="licenseCode" type="string">
  The license code value (example: `"LICENSECODE12345"`).
</ParamField>

<ParamField path="status" type="string">
  License status (example: `"ACTIVE"`).
</ParamField>

<ParamField path="expiredAt" type="string (ISO 8601 datetime)">
  License expiration date (example: `"2025-12-12T19:46:24.000Z"`).
</ParamField>

<ParamField path="transactionId" type="string (UUID)">
  Related transaction ID.
</ParamField>

<ParamField path="productId" type="string (UUID)">
  Associated product ID.
</ParamField>

<ParamField path="customerId" type="string">
  Customer ID linked to this license.
</ParamField>

<ParamField path="customerName" type="string (UUID)">
  Customer’s full name.
</ParamField>

<ParamField path="customerEmail" type="string">
  Customer’s email address.
</ParamField>

<ParamField path="activationLimit" type="string">
  Maximum allowed activations (example: `"Tidak terbatas"` → means “Unlimited”).
</ParamField>

<ParamField path="useCount" type="number">
  Number of times the license has been used.
</ParamField>

<ParamField path="createdAt" type="string (ISO 8601 datetime)">
  License creation timestamp.
</ParamField>

<ParamField path="updatedAt" type="string (ISO 8601 datetime)">
  Last update timestamp.
</ParamField>

<ParamField path="membershipTierId" type="string (UUID)">
  ID of the membership tier associated with the license.
</ParamField>

<ParamField path="param" type="string">
  Name of the membership tier (example: `"Master Black Belt Membership"`).
</ParamField>
